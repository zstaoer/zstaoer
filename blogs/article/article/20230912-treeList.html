        @Override
    public List<Area> treeList() {
        List<Area> dataList = areaMapper.treeList();
        long startTime = System.currentTimeMillis();
        List<Area> result = dataList.stream().filter(area -> area.getPid() == 0).map(parent -> {
            parent.setChild(getChildren(dataList, parent));
            return parent;
        }).collect(Collectors.toList());
        System.out.println(System.currentTimeMillis() - startTime);
        return result;
    }

    private List<Area> getChildren(List<Area> dataList, Area parent) {
        //设置子数据，如果子数据的pid等于父数据的id，这设置值
        List<Area> children = dataList.stream().filter(area -> area.getPid() == parent.getId()).map(p -> {
            p.setChild(getChildren(dataList, p));
            return p;
        }).collect(Collectors.toList());
        return children;
    }

        @Override
    public List<Tree<Long>> treeListFn() {
        long startTime = System.currentTimeMillis();
        List<Area> dataList = areaMapper.treeList();
        List<TreeNode<Long>> data = dataList.stream().map(getNodeFunction()).collect(Collectors.toList());
        List<Tree<Long>> result = TreeUtil.build(data, 0L);
        System.out.println(System.currentTimeMillis() - startTime);
        return result;
    }

    private Function<Area, TreeNode<Long>> getNodeFunction() {
        return area -> {
            TreeNode<Long> node = new TreeNode<>();
            node.setId(Long.valueOf(area.getId()));
            node.setName(area.getCname());
            node.setParentId(Long.valueOf(area.getPid()));
            node.setWeight(area.getCtype());
            Map<String,Object> extra new HashMap();
            extra.put("path",area.getPath());
            extra.put("short",area.getShort());
            return node;
        };
    }

         -- 这种递归方式查询的速度较快，还支持前端懒加载
<select id="treeList" resultType="com.example.demo.pojo.Area">
        WITH RECURSIVE cte as (
            select a.*
            FROM province a
           -- 查所有 可以改为懒加载 where id = #{id}
            where id = 1
            UNION ALL
            select b.*
            FROM province b
            INNER JOIN cte c on c.id = b.pid
        )
        SELECT *
        FROM cte
    </select>
