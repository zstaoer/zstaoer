    1. # 计算属性（computed）

    <script setup>
import { computed, ref } from 'vue'
const list = ref([1, 2, 3, 4, 5, 6, 7, 8])
const computedList = computed(() => {
  return list.value.filter(item => item > 2)
})
const addFn = () => {
list.value.push(666)
}
</script>

<template>
  <div>
    <div>原始数据：{{ list }}</div>
    <div>计算后得数据：{{ computedList }}</div>
    <button @click="addFn">修改</button>
  </div>
</template>

    1. # watch监听

    <script setup>
import { ref, watch } from 'vue'
const count = ref(0)
const nickname = ref('张三')
const changeCount = () => {
  count.value++
}
const changeNickname = () => {
  nickname.value = '李四'
}
//监视单个数据变化
//watch(ref对象,(newValue,oldValue) => {.....})
// watch(count,(newValue,oldValue) => {
//   console.log(newValue,oldValue);
// })

//监视多个数据变化
//watch([ref对象1,ref对象2],(newArr,oldArr) => {.....})
// watch([count, nickname], (newArr, oldArr) => {
//   console.log(newArr, oldArr);
// })
const userInfo = ref({
  name: 'zs',
  age: 18
})
const setUserInfo = () => {
  userInfo.value.age++
}
//immediate:进入页面开始监听
//deep:深度监听，只改变对象一个属性也能监听到
// watch(userInfo, (newValue) => {
//   console.log(newValue);
// }, {
//   //immediate:true,
//   deep: true
// })

//只监听对象中的某个属性
watch(
  () => userInfo.value.age,
  (newValue, oldValue) => console.log(newValue, oldValue)
)

</script>

<template>
  <div>
    {{ count }}
    <button @click="changeCount">改数字</button>
    {{ nickname }}
    <button @click="changeNickname">改昵称</button>
    {{ userInfo }}
    <button @click="setUserInfo">改用户</button>
  </div>
</template>

    1. # 父子组件通信

      父传子 父组件中给子组件绑定属性 子组件中props接收

      子传父 父组件给子组件绑定事件 子组件通过emit触发事件

      还可以使用defineModel实现不需要emit触发

      在子组件使用

      const modelValue = defineModel()

      :value = modelValue

      给子组件绑定事件直接修改 <input type = "text" :value = "modelValue" @input = "e => modelValue = e.target.value">

      父组件中给子组件直接使用v-model指令

      须在配置文件中配置才生效

      plugins: [
  vue({
script:{

defineModel:true
}
}),
 ],

      1. # provide与inject的使用

      • provide是向子孙后代组件传递数据
      • inject是接收数据
      1. 顶层组件
      <script setup>
import CenterCom from '@/components/center-com.vue'
import { provide, ref } from 'vue';
const count = ref(100)

//跨层传递普通数据
provide('them-color', 'pink')
//跨层传递响应式数据
provide('count', count)
//顶层组件修改响应式数据
setTimeout(() => {
  count.value = 500
}, 2000)

//传递一个函数给子孙后代组件，方便子孙后代组件修改响应式数据
provide("setCount", (newValue) => {
  count.value = newValue
})


</script>

<template>
  <h1>顶层组件</h1>
  <CenterCom></CenterCom>
</template>

//接收函数

//调用函数修改数据

}
</script>

<template>

    <button @click="clickFn">修改数据</button>
