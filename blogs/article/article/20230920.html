    # 使用pinia构建仓库和持久化

    安装pinia

    npnm add pinia
    1

    安装持久化插件

    pnpm add pinia-plugin-persistedstate -D
    1

    在stores/user目录下创建user.js

    import { defineStore } from 'pinia'
import { ref } from 'vue'

//用户模块 token setToken removeToken
export const useUserStore = defineStore('big-user', () => {
    const token = ref('')
    const setToken = (newToken) => {
        token.value = newToken
    }
    const removeToken = () => {
        token.value = ''
    }
    return {
        token,
        setToken,
        removeToken
    }
},
    {
        persist: true
    }
)

    axios请求封装 在utils目录下创建request.js

    import axios from 'axios'
import { useUserStore } from '@/stores'
import { ElMessage } from 'element-plus'
import router from '@/router'
const baseURL = 'http://localhost:8080/stu-manager'

const instance = axios.create({
    baseURL,
    timeout: 10000
})

// 添加请求拦截器
instance.interceptors.request.use(
    (config) => {
        const useStore = useUserStore()
        if (useStore.token) {
            config.headers.Authorization = useStore.token
        }
        return config
    },
    (err) => Promise.reject(err)
);

// 添加响应拦截器
instance.interceptors.response.use(
    (res) => {
        if (res.data.code === 0) {
            return res;
        }
        ElMessage.error(res.data.message || '服务异常')
        return Promise.reject(res.data)
    },
    (err) => {
        //特殊情况 401 权限不足，token 过期
        //错误默认情况 只给提示
        if (err.response?.status === 401) {
            router.push('/login')
        }
        ElMessage.error(err.response.data.message || '服务异常')
        return Promise.reject(err);
    });

export default instance
export { baseURL }

    # 路由配置

    import { createRouter, createWebHistory, createWebHashHistory } from 'vue-router'
import { useUserStore } from '@/stores'

// 配置history模式
//  1.history模式：createWebHistory 地址栏不带#    
//  2.hash模式：createWebHashHistory 地址栏带#    
const router = createRouter({
  history: createWebHashHistory(import.meta.env.BASE_URL),
  routes: [
    { path: '/login', component: () => import('@/views/login/Login.vue') },//登录页
    {
      path: '/',
      component: () => import('@/views/layout/LayoutContainer.vue'),
      redirect: '/student/manage',
      children: [
        {
          path: '/student/manage',
          component: () => import('@/views/student/StudentManage.vue')
        }, 
        {
          path: '/clasess/manage',
          component: () => import('@/views/clasess/Clasess.vue')
        }
      ]
    }
  ]
})
// 挂载路由导航守卫
router.beforeEach((to, from, next) => {

  console.log('uri:', to.path);
  if (to.path === '/login') return next()
  const userStore = useUserStore()
  const tokenStr = userStore.token
  if (!tokenStr) return next('/login')
  next()
})
export default router
