    /*获取文档是否存在*/
	void getDocument() throws IOException {
		// 创建get请求
		GetRequest getRequest = new GetRequest("student", "1");
		// 不获取返回的_source的上下文
		getRequest.fetchSourceContext(new FetchSourceContext(false));
		//getRequest.storedFields("_none_");
		// 发送请求
		boolean exists = client.exists(getRequest, RequestOptions.DEFAULT);
		System.out.println("addDocument--" + exists);

		/*获取文档信息*/
		if (exists) {
			// 创建请求
			GetRequest getRequest1 = new GetRequest("student", "1");
			// 发送请求
			GetResponse response = client.get(getRequest1, RequestOptions.DEFAULT);
			System.out.println("source--" + response.getSource());
			System.out.println("getDocument--" + response.getSourceAsString());
		}

		/*更新文档信息*/
		if (exists) {
			// 创建请求
			UpdateRequest updateRequest = new UpdateRequest("student", "1");
			//updateRequest.timeout("1s");
			// 修改内容
			Student stu = new Student(1, "ls",29);
			updateRequest.doc(JSON.toJSONString(stu), XContentType.JSON);
			UpdateResponse response = client.update(updateRequest, RequestOptions.DEFAULT);
			System.out.println("updateDocument--" + response.status());
		}

		/*删除文档信息*/
		if (exists) {
			// 创建请求
			DeleteRequest deleteRequest = new DeleteRequest("student", "1");
			//deleteRequest.timeout("1s");
			// 修改内容
			DeleteResponse response = client.delete(deleteRequest, RequestOptions.DEFAULT);
			System.out.println("deleteDocument--" + response.status());
		}
	}

	/*批量添加文档*/
	void batchAddDocument() throws IOException {
		// 批量请求
		BulkRequest bulkRequest = new BulkRequest();
		bulkRequest.timeout("10s");
		// 创建对象
		ArrayList<Student> stuList = new ArrayList<>();
		stuList.add(new Student(1, "ls",29));
		stuList.add(new Student(1, "ls",29));
		stuList.add(new Student(1, "ls",29));
		stuList.add(new Student(1, "ls",29));
		stuList.add(new Student(1, "ls",29));
		// 添加请求
		for (Student stu : stuList ) {
			bulkRequest.add(new IndexRequest("student")
					.id(stu.getId().toString())
					.source(JSON.toJSONString(stu, XContentType.JSON));
		}
		// 执行请求
		BulkResponse response = client.bulk(bulkRequest, RequestOptions.DEFAULT);
		System.out.println("batchAddDocument--" + response.status());
	}

	/*查询*/
	void search() throws IOException {
		// 创建请求
		SearchRequest searchRequest = new SearchRequest("student");
		// 构建搜索条件
		SearchSourceBuilder searchSourceBuilder = new SearchSourceBuilder();
		TermQueryBuilder termQueryBuilder = QueryBuilders.termQuery("name", "zs");// 精确查询
		searchSourceBuilder.query(termQueryBuilder);
		searchSourceBuilder.timeout(new TimeValue(60, TimeUnit.SECONDS));
		searchRequest.source(searchSourceBuilder);
		// 执行请求
		SearchResponse response = client.search(searchRequest, RequestOptions.DEFAULT);
		// 解析查询结果
		System.out.println("search--getHists--" + JSON.toJSONString(response.getHits()));
		for (SearchHit documentFields : response.getHits()) {
			System.out.println("search--getHist--" + documentFields.getSourceAsMap());
		}
	}
}

    • 编写配置类 绑定队列交换机(也可以在@EnableRabbit的属性绑定)
    @Configuration
public class RabbitConfig {

    @Bean
    public Queue topicQueue1(){
        return new Queue("topicQueue1",true);
    }

    @Bean
    public Queue topicQueue2(){
        return new Queue("topicQueue2",true);
    }

    @Bean
    public TopicExchange topicExchange(){
        return new TopicExchange("topicExchange");
    }

    @Bean
    public Binding topicBinding1(){
        return BindingBuilder.bind(topicQueue1()).to(topicExchange())
                .with("ss007.id.*");
    }

    @Bean
    public Binding topicBinding2(){
        return BindingBuilder.bind(topicQueue2()).to(topicExchange())
                .with("ss007.name.*");
    }
}
