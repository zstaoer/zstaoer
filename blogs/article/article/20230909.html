    多线程导出数据写道不同的sheet

       //多线程分页写出数据到多个sheet（多线程查询，但是单线程写入，easyexcel当前版本不支持多线程写出到excel）
	@GetMapping
    public void exportPage() throws IOException {
        setHeader();
        int pages = 20; // 页数
        Long count = Db.count(Student.class);
        int size = (int) Math.ceil((double) count / pages);//每页条数
        ExecutorService executorService = Executors.newFixedThreadPool(pages);
        Map<Integer, IPage<Student>> map = new HashMap<>();
        for (int i = 0; i < pages; i++) {
            int finalI = i;
            executorService.submit(() -> {
                Page<Student> studentPage = new Page<>();
                studentPage.setCurrent(finalI + 1);
                studentPage.setSize(size);
                IPage<Student> result = Db.page(studentPage, Student.class);
                map.put(finalI, result);
            });
        }
        try (ExcelWriter excelWriter = EasyExcel.write(response.getOutputStream(), Student.class).build();) {
            for (Map.Entry<Integer, IPage<Student>> entry : map.entrySet()) {
                Integer num = entry.getKey();
                IPage<Student> dataList = entry.getValue();
                WriteSheet writeSheet = EasyExcel.writerSheet(num, "Sheet" + num).build();
                excelWriter.write(dataList.getRecords(), writeSheet);
            }
        }

    }

    多线程实现excel的读取

       @PostMapping
    public void importExcel(MultipartFile file) {
        ExecutorService executorService = Executors.newFixedThreadPool(20);
        List<Callable<Object>> tasks = new ArrayList<>();
        //有多少sheet循环多少次,一个线程读取一个sheet
        for (int i = 0; i < 20; i++) {
            int num = i;
            tasks.add(() -> {
                EasyExcel.read(file.getInputStream(), Student.class, new DemoDataListener()).sheet(num).doRead();
                return null;
            });
        }
        try {
            executorService.invokeAll(tasks);
        } catch (InterruptedException e) {
            e.printStackTrace();
        }
    }

    public class DemoDataListener extends ServiceImpl<StudentMapper, Student> implements ReadListener<Student> {

    /**
     * 每隔10000条存储数据库，然后清理list ，方便内存回收
     */
    private static final int BATCH_COUNT = 10000;
    private ExecutorService executorService = Executors.newFixedThreadPool(20);
    //这里可以使用普通的集合，由于多线程条件下需要保证线程安全，所有使用ThreadLocal隔离线程
    private ThreadLocal<ArrayList<Student>> studentList = ThreadLocal.withInitial(ArrayList::new);
    //原子类操作，包装线程安全
    private static AtomicInteger count = new AtomicInteger(1);

    //@Resource
    //private DemoDataListener demoDataListener;
	public DemoDataListener getDemoDataListener(){
        return new DemoDataListener();
    }

    static class SaveTask implements Runnable {
        private List<Student> studentList;
        private DemoDataListener demoDataListener;

        public SaveTask(List<Student> studentList, DemoDataListener demoDataListener) {
            this.studentList = studentList;
            this.demoDataListener = demoDataListener;
        }
        @Override
        public void run() {
            demoDataListener.saveBatch(studentList);
            log.info("第" + count.getAndAdd(1) + "次插入" + studentList.size() + "条数据");
        }
    }

    /**
     * 这个每一条数据解析都会来调用
     *
     * @param student one row value. Is is same as {@link AnalysisContext#readRowHolder()}
     * @param context
     */
    @Override
    public void invoke(Student student, AnalysisContext context) {
        studentList.get().add(student);
        if (studentList.get().size() >= BATCH_COUNT) {
            //多线程读取，单线程插入
            //saveData();
            //多线程读取，多线程插入
            asyncSaveData();
        }
    }

    /**
     * 所有数据解析完成了 都会来调用
     *
     * @param context
     */
    @Override
    public void doAfterAllAnalysed(AnalysisContext context) {
        // 这里也要保存数据，确保最后遗留的数据也存储到数据库
        log.info("一个sheet处理完成！");
        if (studentList.get().size() >= BATCH_COUNT) {
            //多线程读取，单线程插入
            //saveData();
            //多线程读取，多线程插入
            asyncSaveData();
        }
    }

    /**
     * 加上存储数据库(同步程插入)
     */
//    private void saveData() {
//        if (!studentList.get().isEmpty()) {
//            saveBatch(studentList.get(), studentList.get().size());
//            log.info("第" + count.getAndAdd(1) + "次插入" + studentList.get().size() + "条数据");
//            studentList.get().clear();
//        }
//    }

    /**
     * 多线程读取，异步插入
     */
    private void asyncSaveData() {
        if (!studentList.get().isEmpty()) {
            ArrayList<Student> students = (ArrayList<Student>) studentList.get().clone();
            executorService.execute(new SaveTask(students, getDemoDataListener()));
            studentList.get().clear();
        }
    }
}
